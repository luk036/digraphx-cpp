136 for (const auto& entry : _view_items(this->_digraph)) {

289 auto _find_cycle() -> py::Generator<Node> {
290 auto visited = std::unordered_map<Node, Node>{};
291 if constexpr (requires { this->_digraph.size(); }) visited.reserve(this->_digraph.size());
292 for (const auto& entry : _view_items(this->_digraph)) {
293 const auto& vtx = _get_key(entry);
294 if (visited.contains(vtx)) continue;
295 auto utx = vtx;
296 visited[utx] = vtx;
297 while (this->_pred.contains(utx)) {
298 utx = this->_pred[utx].first;
299 auto it = visited.find(utx);
300 if (it != visited.end()) {
301 if (it->second == vtx) co_yield utx;
302 break;
303 }
304 visited[utx] = vtx;
305 }
306 }
307 co_return;
308 }
