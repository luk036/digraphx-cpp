132inline ThreadPool::ThreadPool(size_t threads) : stop(false) {
133 for (size_t i = 0; i < threads; ++i)
134 workers.emplace_back([this] {
135 for (;;) {
136 std::function<void()> task;
137
138 {
139 std::unique_lock<std::mutex> lock(this->queue_mutex);
140 this->condition.wait(lock,
141 [this] { return this->stop || !this->tasks.empty(); });
142 if (this->stop && this->tasks.empty()) return;
143 task = std::move(this->tasks.front());
144 this->tasks.pop();
145 }
146
147 task();
148 }
149 });
150}
